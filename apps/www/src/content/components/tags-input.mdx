## Usage

```tsx
import {
  TagsInput,
  TagsInputClearTrigger,
  TagsInputContext,
  TagsInputControl,
  TagsInputInput,
  TagsInputItem,
  TagsInputLabel,
} from "@/components/ui/tags-input"
```

```tsx
<TagsInput className="w-[60%] space-y-2">
  <TagsInputLabel>Frameworks</TagsInputLabel>
  <TagsInputControl>
    <TagsInputItem index={1} value={1} />
    <TagsInputInput placeholder="Add Framework" />
  </TagsInputControl>
  <TagsInputClearTrigger>Clear</TagsInputClearTrigger>
</TagsInput>
```
