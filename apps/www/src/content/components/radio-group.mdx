## Usage

```tsx
import {
  RadioGroup,
  RadioGroupItem,
  RadioGroupItemControl,
  RadioGroupItemText,
} from "@/components/ui/radio-group"
```

```tsx
<RadioGroup defaultValue="option-one">
  <RadioGroupItem value="option-one">
    <RadioGroupItemControl />
    <RadioGroupItemText>Option One</RadioGroupItemText>
  </RadioGroupItem>
  <RadioGroupItem value="option-two">
    <RadioGroupItemControl />
    <RadioGroupItemText>Option Two</RadioGroupItemText>
  </RadioGroupItem>
</RadioGroup>
```
